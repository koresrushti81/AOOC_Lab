public class ex1_merge_array {
    public static void main(String[] args) {
        int []a={2,3,5,7,9,8};
        int []b={12,2,4,6};
        int []c=new int[a.length+b.length];
        for(int i=0;i<a.length;i++){
            c[i]=a[i];
        }
        for(int i=0;i<b.length;i++){
            c[a.length+i]=b[i];
        }
        for(int i=0;i<c.length;i++){
            System.out.print(" "+c[i]);
        }
    }
}
