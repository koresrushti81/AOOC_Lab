class StaticDemo {
    static int count;

    static {
        count = 10;
        System.out.println("Static Block Executed");
    }

    StaticDemo() {
        count++;
    }

    static void showCount() {
        System.out.println("Count = " + count);
    }

    public static void main(String[] args) {
        StaticDemo s1 = new StaticDemo();
        StaticDemo s2 = new StaticDemo();
        showCount();
    }
}
